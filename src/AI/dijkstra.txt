import heapq

def dijkstra(graph, start):
    # Priority Queue stores: (distance, current_node)
    # We greedily pick the smallest distance first
    pq = [(0, start)]
    distances = {node: float('inf') for node in graph}
    distances[start] = 0
    visited = set()

    while pq:
        # 1. Greedy Step: Pop the node with the smallest distance
        current_distance, current_node = heapq.heappop(pq)

        # 2. If we already found a better path, skip
        if current_node in visited:
            continue
        visited.add(current_node)

        # 3. Look at neighbors
        for neighbor, weight in graph[current_node].items():
            distance = current_distance + weight

            # 4. If this path is shorter, update it
            if distance < distances[neighbor]:
                distances[neighbor] = distance
                heapq.heappush(pq, (distance, neighbor))

    return distances

# Example Graph
graph = {
    'A': {'B': 1, 'C': 4},
    'B': {'A': 1, 'C': 2, 'D': 5},
    'C': {'A': 4, 'B': 2, 'D': 1},
    'D': {'B': 5, 'C': 1}
}

results = dijkstra(graph, 'A')
print("Shortest distances from A:")
for node, dist in results.items():
    print(f"To {node}: {dist}")
