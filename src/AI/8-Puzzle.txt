import heapq

# 0 represents the empty space (the 'hole')
GOAL = (1, 2, 3, 4, 5, 6, 7, 8, 0)

def get_manhattan_dist(state):
    # h(n): Calculates how far each tile is from its target spot
    dist = 0
    for i, tile in enumerate(state):
        if tile != 0:
            # Current position (row, col)
            r, c = divmod(i, 3)
            # Goal position (row, col)
            gr, gc = divmod(GOAL.index(tile), 3)
            dist += abs(r - gr) + abs(c - gc)
    return dist

def get_neighbors(state):
    # Generates all possible moves (Up, Down, Left, Right)
    neighbors = []
    idx = state.index(0)
    r, c = divmod(idx, 3)
    
    for dr, dc in [(-1,0), (1,0), (0,-1), (0,1)]:
        nr, nc = r + dr, c + dc
        if 0 <= nr < 3 and 0 <= nc < 3:
            new_idx = nr * 3 + nc
            new_state = list(state)
            # Swap the empty space with the neighbor tile
            new_state[idx], new_state[new_idx] = new_state[new_idx], new_state[idx]
            neighbors.append(tuple(new_state))
    return neighbors

def solve_8_puzzle(start):
    # Priority Queue: (priority, moves_count, current_state)
    pq = [(get_manhattan_dist(start), 0, start)]
    visited = {start: 0} # Keeps track of visited states and their cost

    while pq:
        f, g, current = heapq.heappop(pq)

        if current == GOAL:
            return f"Solved! Total moves: {g}"

        for neighbor in get_neighbors(current):
            new_g = g + 1
            if neighbor not in visited or new_g < visited[neighbor]:
                visited[neighbor] = new_g
                # f = g + h
                f_score = new_g + get_manhattan_dist(neighbor)
                heapq.heappush(pq, (f_score, new_g, neighbor))

# Starting board (0 is the gap)
start_state = (1, 2, 3, 4, 0, 6, 7, 5, 8)
print(solve_8_puzzle(start_state))
